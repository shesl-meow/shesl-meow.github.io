default .Width -}} {{- $height = $height | default .Height -}} {{- end -}} {{- end -}} {{- $small := .SrcSmall | default $src -}} {{- $srcSmallLookup := replaceRE `^\./` "" (.SrcSmall | default "") {{- end -}} {{- $optimizedSrcset := "" -}} {{- $processable := false -}} {{- if $resource -}} {{- if and (eq $resource.ResourceType "image") (in (slice "jpeg" "jpg" "png" "webp") (lower $largeImg.RelPermalink -}} {{- $srcsetParts := slice -}} {{- range slice 480 768 1024 1366 1600 -}} {{- if le . $resource.Width -}} {{- $variant := $resource.Resize (printf "%dx webp q80" .) -}} {{- (printf "%s %dw" $src $width) -}} {{- end -}} {{- $optimizedSrcset = delimit $srcsetParts ", " -}} {{- end -}} {{- $alt := .Alt | default $src -}} {{- $loading := resources.Get "svg/loading.svg" |
